%\ifblank{#1}{\:\cdot\:}{#1}
\NewDocumentCommand{\DeclareTripledDelimiter}{O{*}mmmmm}{%
	\makeatletter
	\ifcsname @tempDelim \endcsname%
	{}
  	\else%
	  \DeclarePairedDelimiterX{\@tempDelim}[2]{#2}{#4}{\ensuremath{##1 \, \delimsize#3 \, \mathopen{}##2}}
  	\fi%
	\expandafter \csname @tempDelim \endcsname #1{#5}{#6}
	\makeatother	
}
